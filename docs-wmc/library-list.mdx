export const LibraryTable = ({data}) => (
  <div style={{
    border: '1px solid var(--ifm-color-emphasis-200)',
    borderRadius: '6px',
    overflow: 'auto',
    marginBottom: '2rem'
  }}>
    <table style={{
      margin: 0, 
      width: '100%', 
      border: 'none',
      borderCollapse: 'collapse'
    }}>
      <thead>
        <tr style={{backgroundColor: 'var(--ifm-background-surface-color)'}}>
          <th style={{width: '60px', textAlign: 'center', padding: '0.75rem 0.5rem', border: 'none', borderBottom: '1px solid var(--ifm-color-emphasis-200)'}}>No</th>
          <th style={{width: '30%', padding: '0.75rem 1rem', border: 'none', borderBottom: '1px solid var(--ifm-color-emphasis-200)'}}>ライブラリ名称</th>
          <th style={{padding: '0.75rem 1rem', border: 'none', borderBottom: '1px solid var(--ifm-color-emphasis-200)'}}>機能説明</th>
        </tr>
      </thead>
      <tbody>
        {data.map((item, index) => (
          <tr key={index}>
            <td style={{textAlign: 'center', padding: '0.75rem 0.5rem', border: 'none'}}>{item.no}</td>
            <td style={{padding: '0.75rem 1rem', border: 'none'}}><strong>{item.name}</strong></td>
            <td style={{padding: '0.75rem 1rem', border: 'none'}}>{item.description}</td>
          </tr>
        ))}
      </tbody>
    </table>
  </div>
);

export const CategorySection = ({title, data}) => {
  const [isOpen, setIsOpen] = React.useState(false);
  return (
    <div style={{marginBottom: '1rem'}}>
      <button
        onClick={() => setIsOpen(!isOpen)}
        style={{
          width: '100%',
          padding: '0.5rem 0.75rem',
          backgroundColor: 'var(--ifm-background-surface-color)',
          border: '2px solid var(--ifm-color-emphasis-200)',
          borderRadius: '6px',
          cursor: 'pointer',
          display: 'flex',
          justifyContent: 'space-between',
          alignItems: 'center',
          fontSize: '0.95rem',
          fontWeight: '500',
          color: 'var(--ifm-color-emphasis-900)',
          transition: 'border-color 0.2s ease'
        }}
        onMouseEnter={(e) => e.currentTarget.style.borderColor = 'var(--ifm-color-primary)'}
        onMouseLeave={(e) => e.currentTarget.style.borderColor = 'var(--ifm-color-emphasis-200)'}
      >
        <span>{title} ({data.length}件)</span>
        <span style={{fontSize: '0.75rem', transition: 'transform 0.2s ease', transform: isOpen ? 'rotate(180deg)' : 'rotate(0deg)'}}>▼</span>
      </button>
      {isOpen && (
        <div style={{marginTop: '0.75rem'}}>
          <LibraryTable data={data} />
        </div>
      )}
    </div>
  );
};
